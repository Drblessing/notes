# To install node-exporter
# Add yourself to the docker group if not already done
sudo snap connect docker:docker-daemon
sudo groupadd docker
sudo usermod -aG docker $USER

# First, ensure the root filesystem is shared
sudo mount --make-rshared /

# Then run the node-exporter container
sudo docker run -d \
  --name node-exporter \
  --restart unless-stopped \
  --net="host" \
  --pid="host" \
  -v "/:/host:ro,rslave" \
  quay.io/prometheus/node-exporter:latest \
  --path.rootfs=/host
